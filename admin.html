// Load platform settings from Firestore
const settingsDoc = await getDoc(doc(db,'settings','platform'));
if(settingsDoc.exists()){
  const s = settingsDoc.data();
  if(s.ads?.length) ADS = s.ads;
  if(s.hero?.images?.length) HERO_IMGS = s.hero.images;
  if(s.hero?.blur !== undefined) HERO_BLUR = s.hero.blur;
  if(s.hero?.overlay !== undefined) HERO_OVERLAY = s.hero.overlay;
}
